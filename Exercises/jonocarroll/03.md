## R

```r
pair_sort <- function(a, b) {
  if (b < a[1]) {
    c(b, a[1])
  } else if (b < a[2]) {
    c(a[1], b)
  } else {
    a
  }
}

second_smallest <- function(x) {
  Reduce(pair_sort, x, init = Inf)[2]
}

second_smallest(c(1, 2, 3, 4, 5, 6))
#> [1] 2
second_smallest(c(200, 40, 21, 13, 2))
#> [1] 13
second_smallest(c(12, 1, 3, 4, 65, 7, -43, 8, 3, 8, 45, 2))
#> [1] 1
```

## Haskell

```haskell
module SecondSmallest where

pair_sort :: (Int, Int) -> Int -> (Int, Int)
pair_sort (x, y) z
    | z < x = (z, x)
    | z < y = (x, z)
    | otherwise = (x, y)

second_smallest :: [Int] -> Int
second_smallest xs = snd $ foldl pair_sort (maxBound, maxBound) xs

main :: IO()
main = do 
    print $ second_smallest [1, 2, 3, 4, 5, 6]
    print $ second_smallest [200, 40, 21, 13, 2]
    print $ second_smallest [12, 1, 3, 4, 65, 7, -43, 8, 3, 8, 45, 2]
```

## Julia

```julia
pair_sort(a, b) = b < a[1] ? [b, a[1]] : b < a[2] ? [a[1], b] : a
second_smallest(xs) = foldl(pair_sort, xs, init=Inf)[2]

second_smallest([1, 2, 3, 4, 5, 6])
second_smallest([200, 40, 21, 13, 2])
```
